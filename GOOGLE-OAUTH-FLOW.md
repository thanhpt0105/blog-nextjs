// File: lib/auth-options.ts

async signIn({ user, account, profile }) {
  if (account?.provider === 'google') {
    
    // 1ï¸âƒ£ Check if user already exists
    const existingUser = await prisma.user.findUnique({
      where: { email: user.email! },
    });

    if (existingUser) {
      // 2ï¸âƒ£ User exists - update info
      await prisma.user.update({
        where: { email: user.email! },
        data: {
          name: user.name,      // â† Update from Google
          image: user.image,    // â† Update profile picture
        },
      });
      
      // 3ï¸âƒ£ Attach existing role to user object
      user.id = existingUser.id;
      user.role = existingUser.role;  // â† Keep ADMIN/EDITOR/USER
      
    } else {
      // 4ï¸âƒ£ New user - create account
      const newUser = await prisma.user.create({
        data: {
          email: user.email!,
          name: user.name || 'Google User',
          image: user.image,
          role: 'USER',        // â† Default role
          password: null,      // â† No password for OAuth
        },
      });
      
      // 5ï¸âƒ£ Attach new user info
      user.id = newUser.id;
      user.role = newUser.role;

  }
  
  return true;  // âœ… Allow credentials sign-in
}
